class Solution {
    public int trap(int[] height) {
        if (height == null || height.length < 3) {
            return 0;
        }

        int left = 0, right = height.length - 1;
        int leftMax = 0, rightMax = 0;
        int ans = 0;

        while (left < right) {
            // 如果左边的高度较小 那么有右边做保底靠山，那计算左边的雨水就是leftMax - height[left] 有右边比他高这么计算是满足的
            if (height[left] < height[right]) {
                // 如果当前高度大于等于左边最大高度，则更新最大高度
                // 这个位置不可能储水，因为它成了新的左边界
                if (height[left] >= leftMax) {
                    leftMax = height[left];
                } else {
                    // 否则，计算储水量
                    ans += leftMax - height[left];
                }
                left++;
            } else { // 如果右边的高度较小或相等 同理左边做保底
                // 如果当前高度大于等于右边最大高度，则更新最大高度
                // 这个位置不可能储水，因为它成了新的右边界
                if (height[right] >= rightMax) {
                    rightMax = height[right];
                } else {
                    // 否则，计算储水量
                    ans += rightMax - height[right];
                }
                right--;
            }
        }
        return ans;
    }
}
